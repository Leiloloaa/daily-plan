### 学习笔记

**算法题复习**

针对字符串对称匹配使用栈处理

1、'aavsdfb' => 'vsdfb' 对对碰 删除字符串中的所有相邻的重复项

```js
// 解题思路
// 1、定义一个 stack
// 2、如果 stack 有值并且 循环项 等于 stack pop 出的这项 就结束本次循环 不保存
// 3、否则就保存
const removeDuplicates=function (s) {
  let stack =[]
  for (let i of s) {
    let c = null
    if(stack.length && i === (c = stack.pop())) continue
    c && stack.push(c)
    stack.push(i)
  }
  return stack.split('')
}

```

2、判断括号的有效性 一定要按照正确的合并顺序
`[{}]` => true `[}}]` => false

```js
// 解题思路
// 但遇到对称匹配的问题可以使用栈 栈是先进后出 正好是对称的
// 1、对称问题先判断奇偶性
// 2、遇到左括号就将右括号存到 stack 中
// 3、如果是又括号就判断 stack.pop 弹出的是否相等
let isValid = function (s) {
  // 按位与等于 1 就是奇数
  if((s&1) === 1) return false
  let stack = []
  for (let i of s) {
    let c = s[i]
    switch (c) {
      case '(':
        stack.push(')')
        break;
      case '[':
        stack.push(']')
        break; 
      case '{':
        stack.push('}')
        break;
      default:
        if(c !== stack.pop()){
          return false
        }
        break;
    }
  }
  return stack.length === 0
} 
```
